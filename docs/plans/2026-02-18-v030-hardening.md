```typescript
// extensions/subagent/env.ts

const ALLOWED_PREFIXES = ["PI_", "NODE_", "NPM_", "NVM_", "LANG", "LC_", "XDG_"];

const ALLOWED_EXPLICIT = new Set([
  "PATH",
  "HOME",
  "SHELL",
  "TERM",
  "USER",
  "LOGNAME",
  "TMPDIR",
  "EDITOR",
  "VISUAL",
  "SSH_AUTH_SOCK",
  "COLORTERM",
  "FORCE_COLOR",
  "NO_COLOR",
]);

export function buildSubagentEnv(
  extra?: Record<string, string>,
): Record<string, string | undefined> {
  const filtered: Record<string, string | undefined> = {};

  for (const [key, value] of Object.entries(process.env)) {
    if (value === undefined) continue;
    if (ALLOWED_EXPLICIT.has(key) || ALLOWED_PREFIXES.some((p) => key.startsWith(p))) {
      filtered[key] = value;
    }
  }

  // Passthrough escape hatch
  const passthrough = process.env.PI_SUBAGENT_ENV_PASSTHROUGH;
  if (passthrough) {
    for (const name of passthrough.split(",").map((s) => s.trim()).filter(Boolean)) {
      const val = process.env[name];
      if (val !== undefined) filtered[name] = val;
    }
  }

  // Merge explicit extras (e.g., PI_TDD_GUARD_VIOLATIONS_FILE)
  if (extra) {
    Object.assign(filtered, extra);
  }

  return filtered;
}
```

```typescript
// extensions/subagent/timeout.ts
export const DEFAULT_SUBAGENT_TIMEOUT_MS = 600_000; // 10 minutes

export function getSubagentTimeoutMs(agentTimeout?: number): number {
  if (agentTimeout !== undefined && agentTimeout > 0) return agentTimeout;

  const envVal = process.env.PI_SUBAGENT_TIMEOUT_MS;
  if (envVal) {
    const parsed = Number.parseInt(envVal, 10);
    if (Number.isFinite(parsed) && parsed > 0) return parsed;
  }

  return DEFAULT_SUBAGENT_TIMEOUT_MS;
}
```

```typescript
// extensions/subagent/lifecycle.ts
import type { ChildProcess } from "node:child_process";
import { log } from "../logging.js";

export class ProcessTracker {
  private processes = new Set<ChildProcess>();

  get size(): number {
    return this.processes.size;
  }

  add(proc: ChildProcess): void {
    this.processes.add(proc);
  }

  remove(proc: ChildProcess): void {
    this.processes.delete(proc);
  }

  killAll(): void {
    for (const proc of this.processes) {
      if (!proc.killed) {
        proc.kill("SIGTERM");
        log.debug(`Sent SIGTERM to subagent process ${proc.pid}`);
      }
    }
  }
}
```

```typescript
// extensions/subagent/concurrency.ts
export const DEFAULT_SUBAGENT_CONCURRENCY = 6;

export function getSubagentConcurrency(): number {
  const envVal = process.env.PI_SUBAGENT_CONCURRENCY;
  if (envVal) {
    const parsed = Number.parseInt(envVal, 10);
    if (Number.isFinite(parsed)) return Math.max(1, parsed);
  }
  return DEFAULT_SUBAGENT_CONCURRENCY;
}

export class Semaphore {
  private _active = 0;
  private _queue: Array<() => void> = [];

  constructor(private limit: number) {}

  get active(): number { return this._active; }
  get waiting(): number { return this._queue.length; }

  async acquire(): Promise<() => void> {
    if (this._active < this.limit) {
      this._active++;
      return this._createRelease();
    }
    return new Promise<() => void>((resolve) => {
      this._queue.push(() => {
        this._active++;
        resolve(this._createRelease());
      });
    });
  }

  private _createRelease(): () => void {
    let released = false;
    return () => {
      if (released) return;
      released = true;
      this._active--;
      const next = this._queue.shift();
      if (next) next();
    };
  }
}
```
